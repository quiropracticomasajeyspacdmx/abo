    function selChip(el) {

        document.querySelectorAll('.chip').forEach(c => c.classList.remove('on'));

        el.classList.add('on');

        selectedArea = el.innerText;

    }

    function submit() {

        if(!selectedArea) return alert("Seleccione un área legal.");

       

        const record = {

            id: Date.now(),

            date: new Date().toLocaleDateString(),

            name: document.getElementById('nombre').value,

            tel: document.getElementById('tel').value,

            email: document.getElementById('email').value,

            area: selectedArea,

            desc: document.getElementById('desc').value,

            urgency: document.getElementById('urgencia').value

        };

        DB.push(record);

        saveDB(); // GUARDAR EN MEMORIA

       

        alert("✅ Expediente guardado correctamente.");

        resetForm();

        showTab('crm', document.querySelectorAll('.tab')[1]);

    }

    function resetForm() {

        document.getElementById('nombre').value = "";

        document.getElementById('tel').value = "";

        document.getElementById('email').value = "";

        document.getElementById('desc').value = "";

        selectedArea = "";

        document.querySelectorAll('.chip').forEach(c => c.classList.remove('on'));

        go(1);

    }

    // --- LÓGICA DE BASE DE DATOS (PERSISTENCIA) ---

    function saveDB() {

        localStorage.setItem('albertosantos_db', JSON.stringify(DB));

        updateBadge();

    }

    function updateBadge() {

        const badge = document.getElementById('badge');

        if(DB.length > 0) {

            badge.style.display = 'inline-block';

            badge.innerText = DB.length;

            document.getElementById('empty-msg').style.display = 'none';

        } else {

            badge.style.display = 'none';

            document.getElementById('empty-msg').style.display = 'block';

        }

    }

    function renderTable() {

        const tbody = document.getElementById('tbody');

        tbody.innerHTML = "";

       

        DB.slice().reverse().forEach(row => {

            const tr = document.createElement('tr');

            let color = row.urgency === 'alta' ? '#ef4444' : row.urgency === 'media' ? '#f59e0b' : '#10b981';

           

            tr.innerHTML = `

                <td style="color:#64748b">${row.date}</td>

                <td style="font-weight:bold; color:#e2e8f0">${row.name}</td>

                <td><a href="tel:${row.tel}" style="color:#c9a84c; text-decoration:none">📞 ${row.tel}</a></td>

                <td><span style="padding:2px 8px; border-radius:10px; background:#1e293b; font-size:11px border:1px solid #334155">${row.area}</span></td>

                <td><div style="width:10px; height:10px; border-radius:50%; background:${color}; display:inline-block; margin-right:5px"></div></td>

            `;

            tr.onclick = () => alert(`📝 DETALLE:\n\nCliente: ${row.name}\nAsunto: ${row.desc}\nContacto: ${row.email}`);

            tbody.appendChild(tr);

        });

        updateBadge();

    }

    function clearData() {

        if(confirm("⚠️ ¿Está seguro de BORRAR TODOS los expedientes?\nEsta acción no se puede deshacer.")) {

            DB = [];

            saveDB();

            renderTable();

        }

    }
